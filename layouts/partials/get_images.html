{{- $images := slice -}} {{- $bundle := partialCached "gallery-bundle.html" . -}} {{- if $bundle -}} {{- $metaAll := partialCached "gallery-meta.html" . -}} {{- $lookup := $metaAll.bySrc -}} {{- range $image := $bundle.Resources.ByType "image" -}} {{- $meta := index $lookup $image.Name -}} {{- if and $meta (ne ($meta.section | default "") "archive") -}} {{- with $image.Meta -}} {{- with .Date -}} {{- if gt .Year 2000 -}} {{- $images = $images | append (dict "image" $image "Date" .) -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- return $images -}}
